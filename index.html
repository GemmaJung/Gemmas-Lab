    <script>

        let score = 0;

        let timerInterval;

        let timeLeft = 10;


        function toggleMode() {

            const isSenior = document.getElementById('mode-toggle').checked;

            const screen = document.getElementById('kiosk-screen');

            if(isSenior) {

                screen.classList.add('senior-mode');

            } else {

                screen.classList.remove('senior-mode');

            }

        }


        function addScore(val) {

            score += val;

            document.getElementById('frustration-score').innerText = score;

        }


        function handleMissClick(e) {

            // 버튼이 아닌 영역 클릭 시

            if(e.target.id === 'miss-click-overlay') {

                addScore(10);

                alert("정확한 지점을 터치해 주세요. (Too far from target)");

            }

        }


        function startApp(type) {

            nextStep('step-' + type);

            startTimer();

        }


        function nextStep(stepId) {

            document.querySelectorAll('#kiosk-screen > div').forEach(div => div.classList.add('hidden-step'));

            document.getElementById(stepId === 'complete' ? 'step-complete' : stepId).classList.remove('hidden-step');

            

            if(stepId === 'complete') {

                clearInterval(timerInterval);

                const isSenior = document.getElementById('mode-toggle').checked;

                let message = isSenior ? 

                    `당신은 결제까지 ${score}점의 좌절을 겪었습니다. 이것은 누군가에게는 매일의 공포입니다.` : 

                    "개선된 모드에서는 훨씬 빠르군요. 디자인의 힘입니다.";

                document.getElementById('final-report').innerText = message;

            }

        }


        function startTimer() {

            clearInterval(timerInterval);

            timeLeft = 10;

            timerInterval = setInterval(() => {

                timeLeft--;

                const timerEl = document.getElementById('timer');

                if(timerEl) timerEl.innerText = timeLeft + 's';

                

                if(timeLeft <= 0) {

                    clearInterval(timerInterval);

                    alert("시간이 초과되어 처음으로 돌아갑니다. (Session Expired)");

                    location.reload();

                }

            }, 1000);

        }
